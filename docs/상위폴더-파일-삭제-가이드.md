## 📝 삭제 스크립트 (안전 버전)

```powershell
# 상위 폴더로 이동
cd C:\Users\a\MyStoreStory\MY_STORE_STORY

# 안전하게 삭제 가능한 파일만 삭제
Write-Host "가이드 문서 삭제 중..." -ForegroundColor Yellow

$safeToDelete = @(
    "인코딩-손상-예방-가이드.md",
    "인코딩-손상-의심-파일-목록.md",
    "재시작-후-상태-확인-보고서.md",
    "프로젝트-재시작-완벽-가이드.md",
    "fix-encoding.ps1"
)

foreach ($file in $safeToDelete) {
    if (Test-Path $file) {
        Write-Host "  삭제: $file" -ForegroundColor Cyan
        Remove-Item $file -Force
    } else {
        Write-Host "  없음: $file" -ForegroundColor Gray
    }
}

Write-Host "완료!" -ForegroundColor Green
Write-Host ""
Write-Host "⚠️  소스 코드 파일은 수동으로 확인 후 삭제하세요!" -ForegroundColor Yellow
```
